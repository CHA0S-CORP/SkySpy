    label: 'MOD+ Turbulence',

    label: 'SEV+ Turbulence',
    filterFn: (p) => ['SEV', 'EXTRM'].includes(p.turbulence_type),

    label: 'Any Icing',
    filterFn: (p) => p.icing_type && p.icing_type !== 'NEG',

    filterFn: (p) => ['MOD', 'SEV'].includes(p.icing_type),

    label: 'SEV Icing',
    filterFn: (p) => p.icing_type === 'SEV',

    label: 'Wind Shear',
